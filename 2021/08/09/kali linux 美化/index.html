oh-my-zsh

// 先更新一下
apt-get update
apt-get upgrade

// 安装 zsh，查看当前系统使用的 shell
echo $SHELL

// 查看系统是否安装了 zsh
cat /etc/shells

// 用 yum 安装 zsh
yum -y install zsh

// 切换 shell 为 zsh
chsh -s /bin/zsh

// 修改主题
vim ~/.zshrc

// 将 ZSH_THEME 改成 agnoster
// 主题列表——https://github.com/ohmyzsh/ohmyzsh/wiki/themes
ZSH_THEME="agnoster"

// 重启后，查看当前shell
source ~/.zshrc   

// 默认主题
ZSH_THEME="robbyrussell"

// 进入 vim 后
i
Esc
:wq
